# Recursion Compound Interest
def comp(principal, year):
    if(year==1):
        return principal * (1.1)
    else: 
        return comp(principal, year-1) * 1.1

# Recursion Find ZERO:
def find0(L):
    if (len(L) == 0):
        return False
    
    if (L[0]==0):
        return True
    else:
        return find0(L[1:len(L)])

# Recursion SORT:

## 1 FIND MINIMUM
def mini(L):
    mini = L[0]

    for x in L:
        if x<mini:
            mini = x
    return mini


## 2 Recursively sort the list L
def Sort(L):
    if (L == []) or (len(L) == 1):
        return L

    m=mini(L)
    L.remove(m)
    return [m]+Sort(L)

# binarySearch:
def binary_search(n, x, first, last):
    if first > last:
        return False
    
    mid = (first + last) // 2

    if n[mid] == x:
        return True
    elif n[mid] > x:
        return binary_search(n, x, first, mid - 1)
    else:  # n[mid] < x
        return binary_search(n, x, mid + 1, last)
